يرجى إدخال مفتاح الأمان (GitHub Token)

إدارة المنتجات 📦

function login() { const token = document.getElementById('gh-token').value.trim(); if(!token) return alert("الرجاء إدخال المفتاح!"); localStorage.setItem('gh_admin_token', token); document.getElementById('login-screen').style.display = 'none'; document.getElementById('dashboard').style.display = 'block'; fetchData(); } function logout() { localStorage.removeItem('gh_admin_token'); location.reload(); } async function fetchData() { const token = localStorage.getItem('gh_admin_token'); document.getElementById('status-msg').innerText = "جاري تحميل المنتجات..."; try { const res = await fetch(`https://api.github.com/repos/${REPO}/contents/${FILE_PATH}`, { headers: { 'Authorization': `token ${token}` } }); if(!res.ok) throw new Error("المفتاح غير صالح أو لا يملك صلاحيات."); const data = await res.json(); fileSha = data.sha; // חשוב כדי להעלות את העדכון // המרה מקוד מוצפן חזרה לטקסט תקין (תומך בערבית) const content = decodeURIComponent(escape(atob(data.content))); // שליפת המערך מתוך הטקסט של הקובץ const jsonStr = content.substring(content.indexOf('['), content.lastIndexOf(']') + 1); currentProducts = JSON.parse(jsonStr); renderProducts(); document.getElementById('status-msg').innerText = "✅ تم التحميل بنجاح"; } catch (e) { document.getElementById('status-msg').innerText = "❌ خطأ: " + e.message; logout(); } } function renderProducts() { const container = document.getElementById('products-list'); container.innerHTML = ''; currentProducts.forEach((p, i) => { container.innerHTML += `
${p.name}
`; }); } async function saveChanges() { const token = localStorage.getItem('gh_admin_token'); const btn = document.getElementById('save-btn'); btn.innerText = "جاري الحفظ..."; btn.disabled = true; // איסוף הנתונים החדשים מהשדות currentProducts.forEach((p, i) => { p.price = Number(document.getElementById(`price-${i}`).value); p.stock = Number(document.getElementById(`stock-${i}`).value); }); // בניית הקובץ מחדש const newFileContent = `const myProducts = ${JSON.stringify(currentProducts, null, 4)};`; // המרה חזרה לפורמט שגיטהאב דורש (Base64 תומך ערבית) const encodedContent = btoa(unescape(encodeURIComponent(newFileContent))); try { const res = await fetch(`https://api.github.com/repos/${REPO}/contents/${FILE_PATH}`, { method: 'PUT', headers: { 'Authorization': `token ${token}`, 'Content-Type': 'application/json' }, body: JSON.stringify({ message: "Update prices and stock from Admin Panel", content: encodedContent, sha: fileSha }) }); if(!res.ok) throw new Error("فشل الحفظ. تأكد من الصلاحيات."); document.getElementById('status-msg').innerText = "🎉 تم الحفظ والنشر بنجاح! سيتم تحديث الموقع خلال ثوانٍ."; fetchData(); // טוען מחדש לקבלת ה-SHA החדש } catch (e) { alert("خطأ: " + e.message); } finally { btn.innerText = "حفظ التغييرات ونشرها 🚀"; btn.disabled = false; } }
